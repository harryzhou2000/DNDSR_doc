74  std::vector<DNDS::index> &serial2Global, const DNDS::MPIInfo &mpi, DNDS::MPI_int nPart)
75  {
76  serial2Global.resize(partition.size());
77  /****************************************/
78  std::vector<DNDS::index> numberAtLocal(nPart, 0);
79  for (auto r : partition)
80  numberAtLocal[r]++;
81  std::vector<DNDS::index> numberTotal(nPart), numberPrev(nPart);
82  MPI::Allreduce(numberAtLocal.data(), numberTotal.data(), nPart, DNDS::DNDS_MPI_INDEX, MPI_SUM, mpi.comm);
83  MPI_Scan(numberAtLocal.data(), numberPrev.data(), nPart, DNDS::DNDS_MPI_INDEX, MPI_SUM, mpi.comm);
84  std::vector<DNDS::index> numberTotalPlc(nPart + 1);
85  numberTotalPlc[0] = 0;
86  for (DNDS::MPI_int r = 0; r < nPart; r++)
87  numberTotalPlc[r + 1] = numberTotalPlc[r] + numberTotal[r], numberPrev[r] -= numberAtLocal[r];
88  // 2 things here: accumulate total and subtract local from prev
89  /****************************************/
90  numberAtLocal.assign(numberAtLocal.size(), 0);
91  DNDS::index iFill = 0;
92  for (auto r : partition)
93  serial2Global[iFill++] = (numberAtLocal[r]++) + numberTotalPlc[r] + numberPrev[r];
