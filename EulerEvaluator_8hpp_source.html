391  ret.setZero();

395  else if constexpr (model == NS_SA || model == NS_SA_3D)
396  {
397  real d = std::min(dWall[iCell][ig], std::pow(veryLargeReal, 1. / 6.));
398  real cb1 = 0.1355;
399  real cb2 = 0.622;
400  real sigma = 2. / 3.;
401  real cnu1 = 7.1;
402  real cnu2 = 0.7;
403  real cnu3 = 0.9;
404  real cw2 = 0.3;
405  real cw3 = 2;
406  real kappa = 0.41;
407  real rlim = 10;
408  real cw1 = cb1 / sqr(kappa) + (1 + cb2) / sigma;
409 
410  real ct3 = 1.2;
411  real ct4 = 0.5;
412 
413  real pMean, asqrMean, Hmean;
414  real gamma = settings.idealGasProperty.gamma;
415  Gas::IdealGasThermal(UMeanXy(I4), UMeanXy(0), (UMeanXy(Seq123) / UMeanXy(0)).squaredNorm(),
416  gamma, pMean, asqrMean, Hmean);
417  // ! refvalue:
418  real muRef = settings.idealGasProperty.muGas;
419 
420  real nuh = UMeanXy(I4 + 1) * muRef / UMeanXy(0);
421

423  real mufPhy, muf;
424  mufPhy = muf = muEff(UMeanXy, T);
425 
426  real Chi = (UMeanXy(I4 + 1) * muRef / mufPhy);
427  real fnu1 = std::pow(Chi, 3) / (std::pow(Chi, 3) + std::pow(cnu1, 3));
428  real fnu2 = 1 - Chi / (1 + Chi * fnu1);
429 
430  Eigen::Matrix<real, dim, 1> velo = UMeanXy(Seq123) / UMeanXy(0);
431  Eigen::Matrix<real, dim, 1> diffRhoNu = DiffUxy(Seq012, {I4 + 1}) * muRef;
432  Eigen::Matrix<real, dim, 1> diffRho = DiffUxy(Seq012, {0});
433  Eigen::Matrix<real, dim, 1> diffNu = (diffRhoNu - nuh * diffRho) / UMeanXy(0);
434  Eigen::Matrix<real, dim, dim> diffRhoU = DiffUxy(Seq012, Seq123);
435  Eigen::Matrix<real, dim, dim> diffU = (diffRhoU - diffRho * velo.transpose()) / UMeanXy(0);
436 
437  Eigen::Matrix<real, dim, dim> Omega = 0.5 * (diffU.transpose() - diffU);
438  real S = Omega.norm() * std::sqrt(2);
439  real Sbar = nuh / (sqr(kappa) * sqr(d)) * fnu2;
440 
441  real Sh;
442 #ifdef USE_NS_SA_NEGATIVE_MODEL
443  if (Sbar < -cnu2 * S)
444  Sh = S + S * (sqr(cnu2) * S + cnu3 * Sbar) / ((cnu3 - 2 * cnu2) * S - Sbar);
445  else //*negative fix
446 #endif
447  Sh = S + Sbar;
448

450  real g = r + cw2 * (std::pow(r, 6) - r);
451  real fw = g * std::pow((1 + std::pow(cw3, 6)) / (std::pow(g, 6) + std::pow(cw3, 6)), 1. / 6.);
452 
453  real ft2 = ct3 * std::exp(-ct4 * sqr(Chi));
454 #ifdef USE_NS_SA_NEGATIVE_MODEL
455  real D = (cw1 * fw - cb1 / sqr(kappa) * ft2) * sqr(nuh / d); //! modified >>

459  real P = cb1 * (1 - ft2) * Sh * nuh;
460 #endif

462 #ifdef USE_NS_SA_NEGATIVE_MODEL

467  fn = (cn1 + std::pow(Chi, 3)) / (cn1 - std::pow(Chi, 3));

470  }
471 #endif

475  ret.setZero();
476  ret(Seq123) = settings.constMassForce * UMeanXy(0);

481  (UMeanXy(I4 + 1) * fn * muRef + mufPhy) / (UMeanXy(0) * sigma) * diffRho.dot(diffNu) / muRef;
482  // std::cout << "P, D " << P / muRef << " " << D / muRef << " " << diffNu.squaredNorm() << std::endl;
483  if (ret.hasNaN())
484  {

496  DNDS_assert(false);
497  }

499  // ret(Eigen::seq(5, Eigen::last)).setZero();

501  }

503  {

507  ret(Seq123) = settings.constMassForce * UMeanXy(0);
