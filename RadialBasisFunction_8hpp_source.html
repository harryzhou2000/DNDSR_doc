53  inline Eigen::Matrix<real, Eigen::Dynamic, Eigen::Dynamic>
54  FRBFBasis(TIn RiXj, RBFKernelType kernel)
55  {
56  using real = DNDS::real;
57  Eigen::Matrix<real, Eigen::Dynamic, Eigen::Dynamic> NiXj;
58  switch (kernel)
59  {
60  /*****/ // using (modified) distance Kernel
61  case Distance:
62  NiXj = RiXj.array();
63  break;
64  case DistanceA1:
65  NiXj = 1 + RiXj.array();
66  break;
67  case InversedDistanceA1:
68  NiXj = (1 + RiXj.array()).inverse();
69  break;
70  case InversedDistanceA1Compact:
71  NiXj = (0.2 + RiXj.array()).inverse() - 1 / 1.2;
72  NiXj.array() *= (RiXj.array() < 1.0).template cast<real>(); // this kernel is bounded
73  break;
74  case CPC2:
75  /*****/ // using CPC2 Kernel
76  NiXj = (1 - RiXj.array()).square().square() * (RiXj.array() * 4 + 1);
77  NiXj.array() *= (RiXj.array() < 1.0).template cast<real>(); // this kernel is bounded
78  break;
79  case CPC0:
80  /*****/ // using CPC0 Kernel
81  NiXj = (1 - RiXj.array()).square();
82  NiXj.array() *= (RiXj.array() < 1.0).template cast<real>(); // this kernel is bounded
83  break;
84  case Gaussian:
85  /*****/ // using Gaussian kernel
86  NiXj = (-(RiXj.array().square())).exp();
87  break;
88  default:
89  DNDS_assert(false);
90  }
91  return NiXj;
92  }
