71  template <int dim = 3, class TVec, class TeV>
72  inline void EulerGasLeftEigenVector(const TVec &velo, real Vsqr, real H, real a, real gamma, TeV &LeV)
73  {
74  LeV.setZero();
75  real gammaBar = gamma - 1;
76  LeV(0, 0) = H + a / gammaBar * (velo(0) - a);
77  LeV(0, 1) = -velo(0) - a / gammaBar;
78  LeV(0, Eigen::seq(Eigen::fix<2>, Eigen::fix<dim>)) =
79  -velo(Eigen::seq(Eigen::fix<1>, Eigen::fix<dim - 1>));
80  LeV(0, dim + 1) = 1;
81 
82  LeV(1, 0) = -2 * H + 4 / gammaBar * (a * a);
83  LeV(1, Eigen::seq(Eigen::fix<1>, Eigen::fix<dim>)) = velo.transpose() * 2;
84  LeV(1, dim + 1) = -2;
85 
86  LeV(Eigen::seq(Eigen::fix<2>, Eigen::fix<dim>), 0) =
87  velo(Eigen::seq(Eigen::fix<1>, Eigen::fix<dim - 1>)) * (-2 * (a * a) / gammaBar);
88  LeV(Eigen::seq(Eigen::fix<2>, Eigen::fix<dim>), Eigen::seq(Eigen::fix<2>, Eigen::fix<dim>))
89  .diagonal()
90  .setConstant(2 * (a * a) / gammaBar);
91 
92  LeV(dim + 1, 0) = H - a / gammaBar * (velo(0) + a);
93  LeV(dim + 1, 1) = -velo(0) + a / gammaBar;
94  LeV(dim + 1, Eigen::seq(Eigen::fix<2>, Eigen::fix<dim>)) =
95  -velo(Eigen::seq(Eigen::fix<1>, Eigen::fix<dim - 1>));
96  LeV(dim + 1, dim + 1) = 1;
97 
98  LeV *= gammaBar / (2 * a * a);
99  }

736  template <int dim = 3, typename TU, typename TGradU, typename TFlux, typename TNorm>
737  void ViscousFlux_IdealGas(const TU &U, const TGradU &GradU, TNorm norm, bool adiabatic, real gamma, real mu, real k, real Cp, TFlux &Flux)
738  {
739  static const auto Seq01234 = Eigen::seq(Eigen::fix<0>, Eigen::fix<dim + 1>);
740  static const auto Seq012 = Eigen::seq(Eigen::fix<0>, Eigen::fix<dim - 1>);
741  static const auto Seq123 = Eigen::seq(Eigen::fix<1>, Eigen::fix<dim>);
742 
743  Eigen::Vector<real, dim> velo = U(Seq123) / U(0);
744  static const real lambda = -2. / 3.;
745  Eigen::Matrix<real, dim, dim> strainRate = (1.0 / sqr(U(0))) *
746  (U(0) * GradU(Seq012, Seq123) -
747  GradU(Seq012, 0) * Eigen::RowVector<real, dim>(U(Seq123))); // dU_j/dx_i

749  (GradU(Seq012, dim + 1) -

754  real p = (gamma - 1) * (U(dim + 1) - U(0) * 0.5 * vSqr);

756  (U(0) * GradP - p * GradU(Seq012, 0));
