63  return std::make_pair(MPI_Datatype(MPI_LONG_DOUBLE), MPI_int(1));
64 
65  if constexpr (std::is_same_v<T, int8_t>)
66  return std::make_pair(MPI_Datatype(MPI_INT8_T), MPI_int(1));
67  if constexpr (std::is_same_v<T, int16_t>)
68  return std::make_pair(MPI_Datatype(MPI_INT16_T), MPI_int(1));
69  if constexpr (std::is_same_v<T, int32_t>)
70  return std::make_pair(MPI_Datatype(MPI_INT32_T), MPI_int(1));
71  if constexpr (std::is_same_v<T, int64_t>)
72  return std::make_pair(MPI_Datatype(MPI_INT64_T), MPI_int(1));
73 
74  if constexpr (sizeof(T) == 1)
75  return std::make_pair(MPI_Datatype(MPI_UINT8_T), MPI_int(1));
76  else if constexpr (sizeof(T) == 2)
77  return std::make_pair(MPI_Datatype(MPI_UINT16_T), MPI_int(1));
78  else if constexpr (sizeof(T) == 4)
79  return std::make_pair(MPI_Datatype(MPI_UINT32_T), MPI_int(1));
80  else if constexpr (sizeof(T) == 8)
81  return std::make_pair(MPI_Datatype(MPI_UINT64_T), MPI_int(1));
82  else
83  return badReturn;
84  }
85  else if constexpr (std::is_array_v<T>)
86  {
87  std::pair<MPI_Datatype, MPI_int> SizCom = BasicType_To_MPIIntType<std::remove_extent_t<T>>();
88  return std::make_pair(SizCom.first, SizCom.second * std::extent_v<T>);
89  }
